    Estado objetivo:
    ```jsx
    import { StrictMode } from 'react'
    import { createRoot } from 'react-dom/client'
    import { useRegisterSW } from 'virtual:pwa-register/react'
    import './index.css'
    import './styles/animations.css'
    import ErrorBoundary from './components/ErrorBoundary.jsx'
    import App from './App.jsx'
    import UpdateBanner from './components/ui/UpdateBanner.jsx'

    function Root() {
      const {
        needRefresh: [needRefresh],
        updateServiceWorker,
      } = useRegisterSW()

      return (
        <StrictMode>
          <ErrorBoundary>
            <App />
          </ErrorBoundary>
          <UpdateBanner
            needRefresh={needRefresh}
            onUpdate={() => updateServiceWorker(true)}
          />
        </StrictMode>
      )
    }

    createRoot(document.getElementById('root')).render(<Root />)
    ```

    Notas:
    - UpdateBanner se renderiza FUERA de ErrorBoundary y App para que sobreviva a navegacion y errores
    - UpdateBanner se renderiza DENTRO de StrictMode para detectar issues en desarrollo
    - `updateServiceWorker(true)` pasa `true` para recargar la pagina tras activar el SW
    - Mantener los imports de CSS y componentes existentes
