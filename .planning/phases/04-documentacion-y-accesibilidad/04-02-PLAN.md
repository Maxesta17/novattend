    **ProgressBar.jsx** — Agregar 4 atributos al `<div>` contenedor exterior (el que tiene className con `bg-border-light`):

    ```jsx
    const handleKeyDown = onClick ? (e) => {
      if (e.key === 'Enter' || e.key === ' ') {
        e.preventDefault()
        onClick()
      }
    } : undefined

    return (
      <div
        className={containerClasses}
        onClick={onClick}
        onKeyDown={handleKeyDown}
        role={onClick ? 'button' : undefined}
        tabIndex={onClick ? 0 : undefined}
      >
    ```
